---
title: 'Trigger Monitor'
api: 'POST /v1/monitors/{monitorId}/trigger'
---

Manually trigger a monitor to run immediately, outside of its scheduled cron timing.

## Path Parameters

<ParamField path="monitorId" type="integer" required>
  The unique identifier of the monitor to trigger
</ParamField>

## Response

<ResponseField name="monitor" type="object" required>
  The monitor object
  
  <Expandable title="Monitor Object">
    <ResponseField name="id" type="integer" required>
      Unique identifier for the monitor
    </ResponseField>
    
    <ResponseField name="label" type="string">
      Optional label for the monitor
    </ResponseField>
    
    <ResponseField name="method" type="string" required>
      HTTP method
    </ResponseField>
    
    <ResponseField name="url" type="string" required>
      Target URL
    </ResponseField>
    
    <ResponseField name="status" type="string" required>
      Current monitor status
    </ResponseField>
    
    <ResponseField name="enabled" type="boolean" required>
      Whether the monitor is enabled
    </ResponseField>
    
    <ResponseField name="cron" type="string" required>
      Cron schedule expression
    </ResponseField>
    
    <ResponseField name="lastCheckAt" type="string">
      Timestamp of last check (ISO 8601)
    </ResponseField>
    
    <ResponseField name="updatedAt" type="string" required>
      Last update timestamp (ISO 8601)
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="check" type="object">
  The check result from the triggered execution
  
  <Expandable title="Check Object">
    <ResponseField name="id" type="integer" required>
      Check identifier
    </ResponseField>
    
    <ResponseField name="status" type="string" required>
      Check status: `ok`, `error`, `retrying`, `pending`, or `unknown`
    </ResponseField>
    
    <ResponseField name="statusCode" type="integer">
      HTTP status code received
    </ResponseField>
    
    <ResponseField name="responseTimeMs" type="integer">
      Response time in milliseconds
    </ResponseField>
    
    <ResponseField name="errorMessage" type="string">
      Error message if check failed
    </ResponseField>
    
    <ResponseField name="selectionType" type="string">
      Type of value selected by selector
    </ResponseField>
    
    <ResponseField name="selectionValue" type="string">
      Value selected by selector
    </ResponseField>
    
    <ResponseField name="diffChanged" type="boolean" required>
      Whether the response changed from previous check
    </ResponseField>
    
    <ResponseField name="diffKind" type="string">
      Type of difference detected
    </ResponseField>
    
    <ResponseField name="diffSummary" type="string">
      Summary of changes
    </ResponseField>
    
    <ResponseField name="diffDetails" type="string">
      Detailed change information
    </ResponseField>
    
    <ResponseField name="checkedAt" type="string" required>
      Timestamp of check (ISO 8601)
    </ResponseField>
  </Expandable>
</ResponseField>

## Example Request

```bash
curl -X POST http://localhost:8080/v1/monitors/1/trigger
```

## Error Responses

<ResponseField name="400">
  Monitor cannot be triggered - the monitor may be in an invalid state
</ResponseField>

<ResponseField name="404">
  Monitor not found - the specified monitorId does not exist
</ResponseField>
